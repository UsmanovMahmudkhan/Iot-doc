// Advanced filtering algorithm
unsigned int applyFilter(unsigned int newValue, 
                        unsigned int *buffer, 
                        unsigned int currentFiltered) {
    // Moving average with outlier detection
    buffer[bufferIndex] = newValue;
    bufferIndex = (bufferIndex + 1) % FILTER_SIZE;
    
    unsigned long sum = 0;
    for (uint8_t i = 0; i < FILTER_SIZE; i++) {
        sum += buffer[i];
    }
    unsigned int average = sum / FILTER_SIZE;
    
    // Smooth transitions
    int change = abs((int)average - (int)currentFiltered);
    if (change > 50) {
        return (currentFiltered * 7 + average * 3) / 10;
    }
    return average;
}

Haptic Feedback Control

// Directional vibration patterns
void updateHaptics(unsigned int distance) {
    if (distance >= WARN_DISTANCE) {
        analogWrite(VIBE_LEFT_PIN, 0);
        analogWrite(VIBE_RIGHT_PIN, 0);
        return;
    }
    
    // Intensity based on distance
    int vibeLevel = map(distance, 0, WARN_DISTANCE, 255, 100);
    
    // Directional feedback
    if (closestSensor == 0) {
        leftVibe = vibeLevel;
        rightVibe = vibeLevel / 2;
    } else if (closestSensor == 2) {
        leftVibe = vibeLevel / 2;
        rightVibe = vibeLevel;
    }
}

// Shift register control for displays
void displayNumberOnSegments(uint8_t digit1, uint8_t digit2) {
    uint8_t seg1Pattern = SEGMENT_PATTERNS[digit1];
    uint8_t seg2Pattern = SEGMENT_PATTERNS[digit2];
    
    shift1Output = seg1Pattern & 0x7F;
    shift2Output = seg2Pattern & 0x7F;
    
    updateShiftRegisters(shift1Output, shift2Output);
}
